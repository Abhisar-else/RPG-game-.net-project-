    // ── PlayerService ────────────────────────────────────────
    public class PlayerService : IPlayerService
    {
        private readonly IPlayerRepository    _players;
        private readonly IDomainEventDispatcher _dispatcher;
        private readonly ICacheService        _cache;

        public PlayerService(IPlayerRepository players,
                             IDomainEventDispatcher dispatcher,
                             ICacheService cache)
        {
            _players    = players;
            _dispatcher = dispatcher;
            _cache      = cache;
        }

        public async Task<PlayerDto> GetPlayerAsync(Guid playerId, CancellationToken ct = default)
        {
            var cached = await _cache.GetAsync<PlayerDto>($"player:{playerId}", ct);
            if (cached is not null) return cached;

            var player = await RequirePlayerAsync(playerId, ct);
            var dto    = MapToDto(player);
            await _cache.SetAsync($"player:{playerId}", dto, TimeSpan.FromMinutes(5), ct);
            return dto;
        }

        public async Task<PlayerDto> CreatePlayerAsync(CreatePlayerRequest req, CancellationToken ct = default)
        {
            if (await _players.ExistsAsync(req.Username, ct))
                throw new DomainException($"Username '{req.Username}' is already taken.");

            var cls    = Enum.Parse<CharacterClass>(req.Class, ignoreCase: true);
            var player = Player.Create(Guid.NewGuid(), req.Username, req.HeroName, cls);

            await _players.AddAsync(player, ct);
            await _dispatcher.DispatchAsync(player.DomainEvents, ct);
            player.ClearEvents();
            return MapToDto(player);
        }

        public async Task<PlayerDto> RenameHeroAsync(Guid playerId, string newName, CancellationToken ct = default)
        {
            var player = await RequirePlayerAsync(playerId, ct);
            player.RenameHero(newName);
            await _players.UpdateAsync(player, ct);
            await InvalidateCacheAsync(playerId, ct);
            return MapToDto(player);
        }

        public async Task<PlayerDto> ChangeClassAsync(Guid playerId, string className, CancellationToken ct = default)
        {
            var player = await RequirePlayerAsync(playerId, ct);
            var cls    = Enum.Parse<CharacterClass>(className, ignoreCase: true);
            player.ChangeClass(cls);
            await _players.UpdateAsync(player, ct);
            await InvalidateCacheAsync(playerId, ct);
            return MapToDto(player);
        }

        public async Task<PlayerDto> PrestigeAsync(Guid playerId, CancellationToken ct = default)
        {
            var player = await RequirePlayerAsync(playerId, ct);
            player.Prestige();
            await _players.UpdateAsync(player, ct);
            await _dispatcher.DispatchAsync(player.DomainEvents, ct);
            player.ClearEvents();
            await InvalidateCacheAsync(playerId, ct);
            return MapToDto(player);
        }

        public async Task<LeaderboardDto> GetLeaderboardAsync(int count = 50, CancellationToken ct = default)
        {
            var cached = await _cache.GetAsync<LeaderboardDto>("leaderboard:top", ct);
            if (cached is not null) return cached;

            var top = await _players.GetTopPlayersAsync(count, ct);
            var dto = new LeaderboardDto(top.Select((p, i) =>
                new LeaderboardEntryDto(i + 1, p.Id, p.HeroName, p.Class.ToString(),
                    p.Level, p.PrestigeLevel)).ToList());

            await _cache.SetAsync("leaderboard:top", dto, TimeSpan.FromMinutes(2), ct);
            return dto;
        }

        private async Task<Player> RequirePlayerAsync(Guid id, CancellationToken ct)
        {
            return await _players.GetByIdAsync(id, ct)
                   ?? throw new DomainException($"Player {id} not found.");
        }

        private Task InvalidateCacheAsync(Guid id, CancellationToken ct) =>
            _cache.RemoveAsync($"player:{id}", ct);

        internal static PlayerDto MapToDto(Player p)
        {
            var s = p.EffectiveStats;
            return new(p.Id, p.Username, p.HeroName, p.Class.ToString(),
                p.Level, p.XP, XpFormula.XpFor(p.Level),
                p.Gold, p.CurrentHp,
                s.MaxHp, s.Attack, s.Defense, s.Speed,
                s.MagicPower, s.CritChance, s.CritMultiplier,
                p.CurrentZoneId, p.PrestigeLevel);
        }
    }

    // ── CombatService ────────────────────────────────────────
    public class CombatService : ICombatService
    {
        private readonly IPlayerRepository       _players;
        private readonly IEnemyRepository        _enemies;
        private readonly ICombatSessionRepository _sessions;
        private readonly IQuestService           _quests;
        private readonly IDomainEventDispatcher  _dispatcher;
        private readonly IGameHubNotifier        _hub;
        private static readonly Random           _rng = new();

        public CombatService(IPlayerRepository players, IEnemyRepository enemies,
            ICombatSessionRepository sessions, IQuestService quests,
            IDomainEventDispatcher dispatcher, IGameHubNotifier hub)
        {
            _players    = players;
            _enemies    = enemies;
            _sessions   = sessions;
            _quests     = quests;
            _dispatcher = dispatcher;
            _hub        = hub;
        }

        public async Task<CombatStateDto> StartCombatAsync(
            Guid playerId, int zoneId, CancellationToken ct = default)
        {
            var existing = await _sessions.GetActiveForPlayerAsync(playerId, ct);
            if (existing is not null)
                throw new DomainException("Player is already in combat.");

            var player = await RequirePlayerAsync(playerId, ct);
            var enemy  = await _enemies.GetRandomForZoneAsync(zoneId, ct);

            var session = new CombatSession
            {
                Id             = Guid.NewGuid(),
                PlayerId       = playerId,
                EnemyId        = enemy.Id,
                ZoneId         = zoneId,
                EnemyCurrentHp = enemy.Stats.MaxHp,
            };
            await _sessions.AddAsync(session, ct);

            await _hub.NotifyPlayerAsync(playerId, "CombatStarted", new
            {
                session.Id, enemy.Name, enemy.Emoji, enemy.Level,
                EnemyHp = enemy.Stats.MaxHp
            });

            return MapCombatState(session, enemy, player);
        }

        public async Task<CombatStateDto> AttackAsync(
            Guid sessionId, Guid playerId, CancellationToken ct = default)
        {
            var (session, player, enemy) = await LoadCombatContextAsync(sessionId, playerId, ct);

            // Player attacks
            var playerDmg = DamageFormula.Calculate(player.EffectiveStats, enemy.Stats, "attack");
            session.EnemyCurrentHp = Math.Max(0, session.EnemyCurrentHp - playerDmg.Final);
            session.Log.Add(new CombatLogEntry
            {
                Turn      = session.Log.Count + 1,
                ActorName = player.HeroName,
                Action    = "ATTACK",
                Damage    = playerDmg.Final,
                IsCrit    = playerDmg.IsCrit,
                Color     = playerDmg.IsCrit ? "amber" : "orange"
            });

            if (session.EnemyCurrentHp <= 0)
                return await HandleVictoryAsync(session, player, enemy, ct);

            // Enemy counter-attack
            var enemyDmg = DamageFormula.Calculate(enemy.Stats, player.EffectiveStats, "attack");
            player.TakeDamage(enemyDmg.Final);
            session.Log.Add(new CombatLogEntry
            {
                Turn      = session.Log.Count + 1,
                ActorName = enemy.Name,
                Action    = "ATTACK",
                Damage    = enemyDmg.Final,
                IsCrit    = enemyDmg.IsCrit,
                Color     = "red"
            });

            if (player.CurrentHp <= 0)
                return await HandleDefeatAsync(session, player, ct);

            await _sessions.UpdateAsync(session, ct);
            await _players.UpdateAsync(player, ct);
            return MapCombatState(session, enemy, player);
        }

        public async Task<CombatStateDto> FleeAsync(
            Guid sessionId, Guid playerId, CancellationToken ct = default)
        {
            var (session, player, enemy) = await LoadCombatContextAsync(sessionId, playerId, ct);

            // Flee has 60% chance; faster characters flee more reliably
            bool success = _rng.Next(100) < 60 + (player.EffectiveStats.Speed - enemy.Stats.Speed) * 2;
            if (!success)
            {
                var dmg = DamageFormula.Calculate(enemy.Stats, player.EffectiveStats, "flee_penalty");
                player.TakeDamage(dmg.Final);
            }

            session.End(CombatResult.Fled, 0, 0, new List<Guid>());
            await _sessions.UpdateAsync(session, ct);
            await _players.UpdateAsync(player, ct);
            await _dispatcher.DispatchAsync(session.DomainEvents, ct);
            session.ClearEvents();
            return MapCombatState(session, enemy, player);
        }

        public async Task<CombatResultDto> ProcessAutoTickAsync(
            Guid sessionId, CancellationToken ct = default)
        {
            var session = await _sessions.GetByIdAsync(sessionId, ct)
                          ?? throw new DomainException("Session not found.");
            var state = await AttackAsync(sessionId, session.PlayerId, ct);
            if (!state.IsOngoing)
            {
                return new CombatResultDto(
                    session.Result.ToString(), 0, 0, new List<InventorySlotDto>(), false);
            }
            return new CombatResultDto("Ongoing", 0, 0, new List<InventorySlotDto>(), false);
        }

        public async Task<IdleRewardDto> ClaimIdleRewardsAsync(
            Guid playerId, CancellationToken ct = default)
        {
            var player  = await RequirePlayerAsync(playerId, ct);
            var elapsed = DateTime.UtcNow - player.LastLoginAt;

            // Cap idle gains at 8 hours
            var idleMinutes = Math.Min((int)elapsed.TotalMinutes, 480);
            long xp   = idleMinutes * (5 + player.Level);
            long gold  = idleMinutes * (2 + player.Level / 2);
            var  items = RollIdleLoot(idleMinutes);

            var leveledUp = player.GainXP(xp);
            player.AddGold(gold);
            foreach (var item in items) player.AddToInventory(item);

            await _players.UpdateAsync(player, ct);
            await _dispatcher.DispatchAsync(player.DomainEvents, ct);
            player.ClearEvents();

            return new IdleRewardDto(xp, gold,
                items.Select(i => MapItem(i)).ToList(),
                elapsed);
        }

        // ── Private Helpers ───────────────────────────────────

        private async Task<CombatStateDto> HandleVictoryAsync(
            CombatSession session, Player player, Enemy enemy, CancellationToken ct)
        {
            var loot = RollLoot(enemy);
            session.End(CombatResult.Victory, enemy.XpReward, enemy.GoldReward,
                loot.Select(i => i.Id).ToList());

            var leveledUp = player.GainXP(enemy.XpReward);
            player.AddGold(enemy.GoldReward);
            foreach (var item in loot) player.AddToInventory(item);

            await _sessions.UpdateAsync(session, ct);
            await _players.UpdateAsync(player, ct);
            await _dispatcher.DispatchAsync(session.DomainEvents.Concat(player.DomainEvents), ct);
            session.ClearEvents(); player.ClearEvents();

            // Advance quest kill-tracking
            await _quests.UpdateProgressAsync(player.Id, $"kill:{enemy.Name.ToLower()}", 1, ct);

            await _hub.NotifyPlayerAsync(player.Id, "CombatVictory", new
            {
                XpGained = enemy.XpReward, GoldGained = enemy.GoldReward,
                LootCount = loot.Count, leveledUp
            });

            return MapCombatState(session, enemy, player);
        }

        private async Task<CombatStateDto> HandleDefeatAsync(
            CombatSession session, Player player, CancellationToken ct)
        {
            session.End(CombatResult.Defeat, 0, 0, new List<Guid>());
            // Respawn with 30% HP at town
            player.Heal((int)(player.EffectiveStats.MaxHp * 0.30f));
            player.MoveTo(1); // Zone 1 = Town

            await _sessions.UpdateAsync(session, ct);
            await _players.UpdateAsync(player, ct);
            await _dispatcher.DispatchAsync(session.DomainEvents, ct);
            session.ClearEvents();
            return MapCombatState(session, null, player);
        }

        private async Task<(CombatSession, Player, Enemy)> LoadCombatContextAsync(
            Guid sessionId, Guid playerId, CancellationToken ct)
        {
            var session = await _sessions.GetByIdAsync(sessionId, ct)
                          ?? throw new DomainException("Combat session not found.");
            if (session.Result != CombatResult.Ongoing)
                throw new DomainException("Combat is already over.");
            if (session.PlayerId != playerId)
                throw new DomainException("Session does not belong to this player.");

            var player = await RequirePlayerAsync(playerId, ct);
            var enemy  = await _enemies.GetByIdAsync(session.EnemyId, ct)
                         ?? throw new DomainException("Enemy not found.");
            return (session, player, enemy);
        }

        private List<Item> RollLoot(Enemy enemy)
        {
            var items = new List<Item>();
            foreach (var entry in enemy.LootTable)
            {
                if (_rng.NextSingle() <= entry.DropRate)
                {
                    int qty = _rng.Next(entry.MinQty, entry.MaxQty + 1);
                    items.Add(new Item { Id = entry.ItemId });
                }
            }
            return items;
        }

    // ── QuestService ─────────────────────────────────────────
    public class QuestService : IQuestService
    {
        private readonly IPlayerRepository _players;
        private readonly IQuestRepository  _quests;
        private readonly IDomainEventDispatcher _dispatcher;
        private readonly IGameHubNotifier  _hub;

        public QuestService(IPlayerRepository players, IQuestRepository quests,
            IDomainEventDispatcher dispatcher, IGameHubNotifier hub)
        {
            _players    = players;
            _quests     = quests;
            _dispatcher = dispatcher;
            _hub        = hub;
        }

        public async Task<List<QuestDto>> GetAvailableQuestsAsync(
            Guid playerId, CancellationToken ct = default)
        {
            var player = await RequirePlayerAsync(playerId, ct);
            var quests = await _quests.GetAvailableForLevelAsync(player.Level, ct);
            var activeIds = player.Quests
                .Where(q => q.Status == QuestStatus.Active)
                .Select(q => q.QuestId).ToHashSet();

            return quests
                .Where(q => !activeIds.Contains(q.Id) || q.IsRepeatable)
                .Select(q => new QuestDto(q.Id, q.Title, q.Description,
                    q.RequiredLevel, q.XpReward, q.GoldReward, null, q.IsRepeatable))
                .ToList();
        }

        public async Task<List<ActiveQuestDto>> GetActiveQuestsAsync(
            Guid playerId, CancellationToken ct = default)
        {
            var player = await RequirePlayerAsync(playerId, ct);
            return player.Quests
                .Where(q => q.Status == QuestStatus.Active)
                .Select(q => new ActiveQuestDto(
                    q.QuestId,
                    q.Quest?.Title ?? "Unknown",
                    q.Status.ToString(),
                    q.Progress,
                    q.Quest?.TargetCount ?? 1,
                    q.ExpiresAt))
                .ToList();
        }

        public async Task<ActiveQuestDto> AcceptQuestAsync(
            Guid playerId, Guid questId, CancellationToken ct = default)
        {
            var player = await RequirePlayerAsync(playerId, ct);
            var quest  = await _quests.GetByIdAsync(questId, ct)
                         ?? throw new DomainException("Quest not found.");

            if (player.Level < quest.RequiredLevel)
                throw new DomainException($"Level {quest.RequiredLevel} required.");

            var playerQuest = new PlayerQuest
            {
                QuestId    = questId,
                Quest      = quest,
                Status     = QuestStatus.Active,
                Progress   = 0,
                AcceptedAt = DateTime.UtcNow,
                ExpiresAt  = quest.TimeLimit.HasValue
                    ? DateTime.UtcNow + quest.TimeLimit.Value : null
            };
            player.Quests.Add(playerQuest);
            await _players.UpdateAsync(player, ct);

            var e = new Domain.Events.QuestAccepted(playerId, questId);
            await _dispatcher.DispatchAsync(new[] { e }, ct);

            return new ActiveQuestDto(questId, quest.Title, "Active", 0, quest.TargetCount, playerQuest.ExpiresAt);
        }

        public async Task<QuestRewardDto> CompleteQuestAsync(
            Guid playerId, Guid questId, CancellationToken ct = default)
        {
            var player = await RequirePlayerAsync(playerId, ct);
            var pq     = player.Quests.FirstOrDefault(q => q.QuestId == questId
                         && q.Status == QuestStatus.Active)
                         ?? throw new DomainException("Quest not active.");
            var quest  = pq.Quest ?? await _quests.GetByIdAsync(questId, ct)
                         ?? throw new DomainException("Quest not found.");

            if (pq.Progress < quest.TargetCount)
                throw new DomainException("Quest objectives not complete.");

            pq.Status = QuestStatus.Completed;
            var leveledUp = player.GainXP(quest.XpReward);
            player.AddGold(quest.GoldReward);

            await _players.UpdateAsync(player, ct);

            var ev = new Domain.Events.QuestCompleted(playerId, questId, quest.XpReward, quest.GoldReward);
            await _dispatcher.DispatchAsync(player.DomainEvents.Append(ev), ct);
            player.ClearEvents();

            await _hub.NotifyPlayerAsync(playerId, "QuestCompleted",
                new { quest.Title, quest.XpReward, quest.GoldReward });

            return new QuestRewardDto(questId, quest.XpReward, quest.GoldReward, null, leveledUp);
        }

        public async Task AbandonQuestAsync(Guid playerId, Guid questId, CancellationToken ct = default)
        {
            var player = await RequirePlayerAsync(playerId, ct);
            var pq     = player.Quests.FirstOrDefault(q => q.QuestId == questId)
                         ?? throw new DomainException("Quest not found.");
            pq.Status = QuestStatus.Failed;
            await _players.UpdateAsync(player, ct);
        }

        public async Task UpdateProgressAsync(Guid playerId, string trackingKey,
            int delta = 1, CancellationToken ct = default)
        {
            var player = await RequirePlayerAsync(playerId, ct);
            foreach (var pq in player.Quests.Where(q => q.Status == QuestStatus.Active))
            {
                var objective = pq.Quest?.Objectives
                    .FirstOrDefault(o => o.TrackingKey == trackingKey);
                if (objective is null) continue;
                pq.Progress = Math.Min(pq.Progress + delta, pq.Quest!.TargetCount);
            }
            await _players.UpdateAsync(player, ct);
        }

        private async Task<Player> RequirePlayerAsync(Guid id, CancellationToken ct) =>
            await _players.GetByIdAsync(id, ct) ?? throw new DomainException("Player not found.");
    }

    // ── SaveLoadService ──────────────────────────────────────
    public class SaveLoadService : ISaveLoadService
    {
        private readonly IPlayerRepository  _players;
        private readonly ISaveGameRepository _saves;
        private readonly System.Text.Json.JsonSerializerOptions _jsonOpts = new()
        {
            PropertyNamingPolicy = System.Text.Json.JsonNamingPolicy.CamelCase,
            WriteIndented = false
        };

        public SaveLoadService(IPlayerRepository players, ISaveGameRepository saves)
        {
            _players = players;
            _saves   = saves;
        }

        public async Task<SaveGame> SaveAsync(Guid playerId, string slotName,
            string? label = null, CancellationToken ct = default)
        {
            var player = await _players.GetByIdAsync(playerId, ct)
                         ?? throw new DomainException("Player not found.");
            var slot   = Enum.Parse<SaveSlot>(slotName, ignoreCase: true);

            var snapshot = System.Text.Json.JsonSerializer.Serialize(player, _jsonOpts);
            var save = new SaveGame
            {
                Id          = Guid.NewGuid(),
                PlayerId    = playerId,
                Slot        = slot,
                Snapshot    = snapshot,
                Label       = label ?? $"Level {player.Level} — {DateTime.UtcNow:g}",
                SaveVersion = 1,
                SavedAt     = DateTime.UtcNow
            };

            await _saves.UpsertAsync(save, ct);
            return save;
        }

        public async Task<PlayerDto> LoadAsync(Guid playerId, string slotName, CancellationToken ct = default)
        {
            var slot = Enum.Parse<SaveSlot>(slotName, ignoreCase: true);
            var save = await _saves.GetAsync(playerId, slot, ct)
                       ?? throw new DomainException($"No save in slot {slotName}.");

            var player = System.Text.Json.JsonSerializer.Deserialize<Player>(save.Snapshot, _jsonOpts)
                         ?? throw new DomainException("Corrupt save data.");

            await _players.UpdateAsync(player, ct);
            return PlayerService.MapToDto(player);
        }

        public async Task<List<SaveMetaDto>> ListSavesAsync(Guid playerId, CancellationToken ct = default)
        {
            var saves = await _saves.GetAllForPlayerAsync(playerId, ct);
            return saves.Select(s => new SaveMetaDto(
                s.Slot.ToString(), s.Label,
                // Peek level from JSON cheaply
                0, s.SavedAt)).ToList();
        }

        public Task DeleteSaveAsync(Guid playerId, string slotName, CancellationToken ct = default)
        {
            var slot = Enum.Parse<SaveSlot>(slotName, ignoreCase: true);
            return _saves.DeleteAsync(playerId, slot, ct);
        }
    }
