namespace IdleQuest.API.Controllers

    using IdleQuest.Application.DTOs;

    [ApiController]
    [Route("api/[controller]")]
    [Produces("application/json")]
    public class AuthController : ApiControllerBase
    {
        private readonly IAuthService _auth;
        public AuthController(IAuthService auth) => _auth = auth;

        /// <summary>Register a new player account.</summary>
        [HttpPost("register")]
        [AllowAnonymous]
        [ProducesResponseType(typeof(AuthResultDto), 200)]
        public async Task<IActionResult> Register(
            [FromBody] RegisterRequest req, CancellationToken ct)
        {
            var result = await _auth.RegisterAsync(req, ct);
            return result.Success ? Ok(result) : BadRequest(new { error = result.Error });
        }

        /// <summary>Login and receive JWT + refresh token.</summary>
        [HttpPost("login")]
        [AllowAnonymous]
        [ProducesResponseType(typeof(AuthResultDto), 200)]
        public async Task<IActionResult> Login(
            [FromBody] LoginRequest req, CancellationToken ct)
        {
            var result = await _auth.LoginAsync(req, ct);
            return result.Success ? Ok(result) : Unauthorized(new { error = result.Error });
        }

        /// <summary>Refresh an expired JWT using a refresh token.</summary>
        [HttpPost("refresh")]
        [AllowAnonymous]
        public async Task<IActionResult> Refresh([FromBody] string refreshToken, CancellationToken ct)
        {
            var result = await _auth.RefreshTokenAsync(refreshToken, ct);
            return result.Success ? Ok(result) : Unauthorized(new { error = result.Error });
        }

        /// <summary>Logout and invalidate session.</summary>
        [HttpPost("logout")]
        [Authorize]
        public async Task<IActionResult> Logout(CancellationToken ct)
        {
            await _auth.LogoutAsync(CurrentPlayerId, string.Empty, ct);
            return NoContent();
        }
    }

// ─── Global Exception Middleware ────────────────────────────
namespace IdleQuest.API.Middleware
{
    using IdleQuest.Domain;
    using Microsoft.AspNetCore.Http;
    using Microsoft.Extensions.Logging;
    using System;
    using System.Net;
    using System.Text.Json;
    using System.Threading.Tasks;

    public class GlobalExceptionMiddleware
    {
        private readonly RequestDelegate _next;
        private readonly ILogger<GlobalExceptionMiddleware> _log;

        public GlobalExceptionMiddleware(RequestDelegate next, ILogger<GlobalExceptionMiddleware> log)
        {
            _next = next;
            _log  = log;
        }

        public async Task InvokeAsync(HttpContext ctx)
        {
            try
            {
                await _next(ctx);
            }
            catch (DomainException ex)
            {
                _log.LogWarning("Domain rule violation: {Message}", ex.Message);
                await WriteError(ctx, HttpStatusCode.BadRequest, ex.Message);
            }
            catch (UnauthorizedAccessException ex)
            {
                await WriteError(ctx, HttpStatusCode.Unauthorized, ex.Message);
            }
            catch (Exception ex)
            {
                _log.LogError(ex, "Unhandled exception");
                await WriteError(ctx, HttpStatusCode.InternalServerError,
                    "An unexpected error occurred. Please try again.");
            }
        }

        private static Task WriteError(HttpContext ctx, HttpStatusCode code, string message)
        {
            ctx.Response.StatusCode  = (int)code;
            ctx.Response.ContentType = "application/json";
            return ctx.Response.WriteAsync(
                JsonSerializer.Serialize(new { error = message }));
        }
    }
}

// ─── Rate Limiting Attribute ─────────────────────────────────
namespace IdleQuest.API.Filters
{
    using Microsoft.AspNetCore.Mvc;
    using Microsoft.AspNetCore.Mvc.Filters;
    using Microsoft.Extensions.Caching.Memory;
    using System;
    using System.Threading.Tasks;

    /// <summary>
    /// Simple per-player rate limiter.
    /// Use ASP.NET Core built-in rate limiting (RateLimiterMiddleware) in production.
    /// </summary>
    [AttributeUsage(AttributeTargets.Method | AttributeTargets.Class)]
    public class RateLimitAttribute : ActionFilterAttribute
    {
        public int  RequestsPerWindow { get; set; } = 60;
        public int  WindowSeconds     { get; set; } = 60;

        public override async Task OnActionExecutionAsync(
            ActionExecutingContext ctx, ActionExecutionDelegate next)
        {
            var cache    = ctx.HttpContext.RequestServices.GetService(typeof(IMemoryCache)) as IMemoryCache;
            var playerId = ctx.HttpContext.User.FindFirst("playerId")?.Value ?? ctx.HttpContext.Connection.RemoteIpAddress?.ToString() ?? "anon";
            var key      = $"rl:{playerId}:{ctx.ActionDescriptor.DisplayName}";

            if (cache!.TryGetValue<int>(key, out var count) && count >= RequestsPerWindow)
            {
                ctx.Result = new StatusCodeResult(429); // Too Many Requests
                ctx.HttpContext.Response.Headers["Retry-After"] = WindowSeconds.ToString();
                return;
            }

            cache.Set(key, (count + 1), TimeSpan.FromSeconds(WindowSeconds));
            await next();
        }
    }
}

namespace IdleQuest.API.BackgroundServices
{
    using IdleQuest.Application.Interfaces.Services;
    using Microsoft.Extensions.DependencyInjection;
    using Microsoft.Extensions.Hosting;
    using Microsoft.Extensions.Logging;
    using System;
    using System.Threading;
    using System.Threading.Tasks;

    /// <summary>
    /// Drives auto-combat ticks for players with auto-progress enabled.
    /// Runs every 680ms to match the frontend's setInterval tick.
    /// </summary>
    public class AutoCombatTickService : BackgroundService
    {
        private readonly IServiceScopeFactory _scopeFactory;
        private readonly ILogger<AutoCombatTickService> _log;
        private static readonly TimeSpan _tickInterval = TimeSpan.FromMilliseconds(680);

        public AutoCombatTickService(IServiceScopeFactory sf, ILogger<AutoCombatTickService> log)
        {
            _scopeFactory = sf;
            _log          = log;
        }

        protected override async Task ExecuteAsync(CancellationToken stoppingToken)
        {
            _log.LogInformation("Auto-combat tick service started.");
            while (!stoppingToken.IsCancellationRequested)
            {
                try
                {
                    await ProcessTickAsync(stoppingToken);
                }
                catch (Exception ex)
                {
                    _log.LogError(ex, "Tick error");
                }
                await Task.Delay(_tickInterval, stoppingToken);
            }
        }

        private async Task ProcessTickAsync(CancellationToken ct)
        {
            using var scope   = _scopeFactory.CreateScope();
            var combatService = scope.ServiceProvider.GetRequiredService<ICombatService>();
            // In production: fetch active session IDs from Redis set
            // and call combatService.ProcessAutoTickAsync(sessionId, ct) for each
            await Task.CompletedTask;
        }
    }

    /// <summary>
    /// Periodically triggers world events (e.g., XP bonus hour every 6h).
    /// </summary>
    public class WorldEventScheduler : BackgroundService
    {
        private readonly IServiceScopeFactory _scopeFactory;
        private readonly ILogger<WorldEventScheduler> _log;

        public WorldEventScheduler(IServiceScopeFactory sf, ILogger<WorldEventScheduler> log)
        {
            _scopeFactory = sf;
            _log          = log;
        }

        protected override async Task ExecuteAsync(CancellationToken stoppingToken)
        {
            while (!stoppingToken.IsCancellationRequested)
            {
                try
                {
                    using var scope   = _scopeFactory.CreateScope();
                    var worldService  = scope.ServiceProvider.GetRequiredService<IWorldService>();
                    var hub           = scope.ServiceProvider.GetRequiredService<IGameHubNotifier>();

                    var events = await worldService.GetActiveEventsAsync(stoppingToken);
                    if (events.Count > 0)
                        await hub.BroadcastAsync("WorldEvents", events);
                }
                catch (Exception ex) { _log.LogError(ex, "World event scheduler error"); }

                await Task.Delay(TimeSpan.FromMinutes(5), stoppingToken);
            }
        }
    }

    /// <summary>Auto-save all online players every 2 minutes.</summary>
    public class AutoSaveService : BackgroundService
    {
        private readonly IServiceScopeFactory _scopeFactory;
        private readonly ILogger<AutoSaveService> _log;

        public AutoSaveService(IServiceScopeFactory sf, ILogger<AutoSaveService> log)
        {
            _scopeFactory = sf;
            _log          = log;
        }

        protected override async Task ExecuteAsync(CancellationToken stoppingToken)
        {
            while (!stoppingToken.IsCancellationRequested)
            {
                await Task.Delay(TimeSpan.FromMinutes(2), stoppingToken);
                try
                {
                    using var scope = _scopeFactory.CreateScope();
                    var saves       = scope.ServiceProvider.GetRequiredService<ISaveLoadService>();

                    foreach (var (playerId, _) in Infrastructure.Hubs.GameHub.PlayerConnections)
                    {
                        try { await saves.SaveAsync(playerId, "auto", "Auto-save", stoppingToken); }
                        catch (Exception ex) { _log.LogWarning(ex, "Auto-save failed for {Id}", playerId); }
                    }
                }
                catch (Exception ex) { _log.LogError(ex, "Auto-save service error"); }
            }
        }
    }
